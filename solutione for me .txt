    // calculate reserved seats
    private static int getCount(int count, boolean set11, boolean set12, boolean set21, boolean set22, boolean set31) {

            count++;

        if (set12) {
            count++;
        }
        if (set21) {
            count++;
        }
        if (set22) {
            count++;
        }
        if (set31) {
            count++;
        }
        return count;
    }
